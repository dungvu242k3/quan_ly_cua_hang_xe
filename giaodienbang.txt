  <script>
    lucide.createIcons();

    // Dữ liệu bao gồm tên và số lượng tương ứng
    const dataConfig = {
      name: [
        { label: "Nguyễn Văn A", count: 2 },
        { label: "Trần Thị B", count: 2 },
        { label: "Phạm Thị D", count: 1 },
        { label: "Lê Văn C", count: 5 }
      ],
      type: [
        { label: "Thử việc", count: 45 },
        { label: "Chính thức", count: 120 },
        { label: "Cộng tác viên", count: 12 },
        { label: "Hợp đồng khoán", count: 5 }
      ],
      status: [
        { label: "Có hiệu lực", count: 165 },
        { label: "Đã thanh lý", count: 15 },
        { label: "Sắp hết hạn", count: 8 }
      ]
    };

    // Render danh sách checkbox kèm số lượng
    document.querySelectorAll('.dropdown-container').forEach(container => {
      const type = container.dataset.filter;
      const listContainer = container.querySelector('.list-container');
      listContainer.innerHTML = dataConfig[type].map(item => `
        <label class="flex items-center justify-between px-2 py-2 hover:bg-slate-50 rounded-lg cursor-pointer transition-colors">
          <div class="flex items-center space-x-2.5">
            <input type="checkbox" value="${item.label}" class="filter-checkbox rounded border-slate-300 text-primary-500 w-4 h-4">
            <span class="text-[13px] text-slate-700">${item.label}</span>
          </div>
          <span class="text-[12px] text-slate-500 font-medium">${item.count}</span>
        </label>
      `).join('');
    });

    // Mở/Đóng dropdown
    document.addEventListener('click', (e) => {
      const btn = e.target.closest('.dropdown-btn');
      if (btn) {
        const menu = btn.nextElementSibling;
        const icon = btn.querySelector('[data-lucide="chevron-down"]');
        document.querySelectorAll('.dropdown-menu').forEach(m => m !== menu && m.classList.add('hidden'));
        menu.classList.toggle('hidden');
        icon.classList.toggle('rotate-180');
      } else if (!e.target.closest('.dropdown-menu')) {
        document.querySelectorAll('.dropdown-menu').forEach(m => m.classList.add('hidden'));
        document.querySelectorAll('.dropdown-btn i').forEach(i => i.classList.remove('rotate-180'));
      }
    });

    // Logic lọc bảng
    function filterTable() {
      const search = document.getElementById('mainSearch').value.toLowerCase();
      const rows = document.querySelectorAll('#tableBody tr');
      const filters = {};

      document.querySelectorAll('.dropdown-container').forEach(c => {
        const type = c.dataset.filter;
        const checked = Array.from(c.querySelectorAll('.filter-checkbox:checked')).map(i => i.value);
        if (checked.length > 0) filters[type] = checked;
      });

      rows.forEach(row => {
        const matchesSearch = row.innerText.toLowerCase().includes(search);
        const matchesName = !filters.name || filters.name.includes(row.dataset.name);
        const matchesType = !filters.type || filters.type.includes(row.dataset.type);
        const matchesStatus = !filters.status || filters.status.includes(row.dataset.status);

        row.style.display = (matchesSearch && matchesName && matchesType && matchesStatus) ? '' : 'none';
      });
    }

    document.getElementById('mainSearch').addEventListener('input', filterTable);
    document.querySelectorAll('.filter-checkbox').forEach(cb => cb.addEventListener('change', filterTable));
    
    document.querySelectorAll('.select-all').forEach(btn => {
      btn.onclick = () => {
        btn.closest('.dropdown-menu').querySelectorAll('.filter-checkbox').forEach(c => c.checked = true);
        filterTable();
      };
    });

    document.querySelectorAll('.clear-all').forEach(btn => {
      btn.onclick = () => {
        btn.closest('.dropdown-menu').querySelectorAll('.filter-checkbox').forEach(c => c.checked = false);
        filterTable();
      };
    });

    function exportExcel() { alert("Đang chuẩn bị tệp tải xuống..."); }
  </script>
