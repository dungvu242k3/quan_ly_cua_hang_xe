<script>
    // 1. DỮ LIỆU GỐC (Đã thêm trạng thái khác để test lọc)
    let rawData = [
        { id: 1, code: 'HS-2025-01', name: 'Hồ sơ nhân sự quý I/2025', doc: 'Quyết định ban hành nội quy công ty', dept: 'Phòng nhân sự', auth: 'Chưa cấu hình', expire: '31/12/2030', status: 'Đang hoạt động', isPinned: false },
        { id: 2, code: 'HS-2025-02', name: 'Hồ sơ công văn đến/đi', doc: 'Công văn đề nghị cung cấp hồ sơ', dept: 'Phòng hành chính', auth: 'Chưa cấu hình', expire: '31/12/2030', status: 'Đang hoạt động', isPinned: false },
        { id: 3, code: 'HS-2024-12', name: 'Hồ sơ nội quy công ty', doc: 'Quyết định ban hành nội quy công ty', dept: 'Phòng hành chính', auth: 'Chưa cấu hình', expire: '—', status: 'Đang hoạt động', isPinned: false },
        { id: 4, code: 'HS-2025-03', name: 'Hồ sơ dự án đầu tư', doc: 'Tờ trình đề xuất mua sắm thiết bị văn phòng', dept: 'Phòng kỹ thuật', auth: 'Chưa cấu hình', expire: '31/12/2030', status: 'Tạm khóa', isPinned: false }
    ];

    let currentTab = 'all';

    // 2. HÀM RENDER (Giữ nguyên label blue-100 cho trạng thái)
    function render(data) {
        const tbody = document.getElementById('data-table');
        tbody.innerHTML = '';

        data.forEach(item => {
            const tr = document.createElement('tr');
            tr.className = "hover:bg-slate-50/80 transition-colors";
            
            // Xử lý màu sắc label dựa trên nội dung (Ví dụ: Tạm khóa dùng màu khác nếu muốn, mặc định vẫn là blue-100 của bạn)
            const statusStyle = item.status === 'Tạm khóa' 
                ? 'bg-slate-100 text-slate-500 border-slate-200' 
                : 'bg-status-blue text-status-text border-blue-100';

            tr.innerHTML = `
                <td class="px-4 py-4 text-center"><input class="row-cb rounded border-slate-300 text-primary size-4" type="checkbox"/></td>
                <td class="px-4 py-4">
                    <div class="flex flex-col">
                        <span class="text-slate-400 text-[11px] mb-0.5">${item.code}</span>
                        <span class="text-slate-900 font-semibold">${item.name}</span>
                    </div>
                </td>
                <td class="px-4 py-4 text-slate-600">${item.doc}</td>
                <td class="px-4 py-4">${item.dept}</td>
                <td class="px-4 py-4 text-slate-400 italic">${item.auth}</td>
                <td class="px-4 py-4 text-slate-600">${item.expire}</td>
                <td class="px-4 py-4">
                    <span class="${statusStyle} px-3 py-1 rounded-full text-[11px] font-semibold border">
                        ${item.status}
                    </span>
                </td>
                <td class="px-4 py-4">
                    <div class="flex items-center justify-center gap-4">
                        <button onclick="togglePin(${item.id})" class="text-slate-400 hover:text-amber-500">
                            <span class="material-symbols-outlined text-[18px] ${item.isPinned ? 'is-pinned' : ''}">push_pin</span>
                        </button>
                        <button class="text-primary hover:text-blue-700"><span class="material-symbols-outlined text-[18px]">edit_square</span></button>
                        <button onclick="deleteRow(${item.id})" class="text-red-500 hover:text-red-700"><span class="material-symbols-outlined text-[18px]">delete</span></button>
                    </div>
                </td>
            `;
            tbody.appendChild(tr);
        });
        document.getElementById('view-count').innerText = data.length;
        document.getElementById('total-count').innerText = rawData.length;
    }

    // 3. LOGIC LỌC ĐA ĐIỀU KIỆN
    function runFilter() {
        const searchVal = document.getElementById('inp-search').value.toLowerCase();
        const selectedDocs = Array.from(document.querySelectorAll('#dropdown-doc .filter-cb:checked')).map(c => c.value);
        const selectedDepts = Array.from(document.querySelectorAll('#dropdown-dept .filter-cb:checked')).map(c => c.value);
        const selectedStatus = Array.from(document.querySelectorAll('#dropdown-status .filter-cb:checked')).map(c => c.value);

        const filtered = rawData.filter(item => {
            const matchSearch = item.name.toLowerCase().includes(searchVal) || item.code.toLowerCase().includes(searchVal);
            const matchDoc = selectedDocs.length === 0 || selectedDocs.includes(item.doc);
            const matchDept = selectedDepts.length === 0 || selectedDepts.includes(item.dept);
            const matchStatus = selectedStatus.length === 0 || selectedStatus.includes(item.status);
            const matchTab = currentTab === 'all' || (currentTab === 'pin' && item.isPinned);

            return matchSearch && matchDoc && matchDept && matchStatus && matchTab;
        });

        render(filtered);
    }

    // 4. CÁC HÀM TIỆN ÍCH
    function toggleDropdown(id) {
        document.querySelectorAll('.dropdown-menu').forEach(d => { if(d.id !== id) d.classList.remove('show') });
        document.getElementById(id).classList.toggle('show');
    }

    function checkAllInDropdown(source, parentId) {
        document.querySelectorAll(`#${parentId} .filter-cb`).forEach(c => c.checked = source.checked);
        runFilter();
    }

    function clearInDropdown(parentId) {
        document.querySelectorAll(`#${parentId} input[type="checkbox"]`).forEach(c => c.checked = false);
        runFilter();
    }

    function togglePin(id) {
        const item = rawData.find(i => i.id === id);
        if(item) item.isPinned = !item.isPinned;
        runFilter();
    }

    function deleteRow(id) {
        if(confirm('Xác nhận xóa?')) {
            rawData = rawData.filter(i => i.id !== id);
            runFilter();
        }
    }

    function changeTab(tab) {
        currentTab = tab;
        const btnAll = document.getElementById('tab-all');
        const btnPin = document.getElementById('tab-pin');
        const activeClass = "flex items-center gap-2 px-4 py-2 bg-white rounded-t-lg border border-b-0 border-slate-200 text-primary font-medium text-[13px] relative after:absolute after:bottom-[-1px] after:left-0 after:right-0 after:h-[2px] after:bg-primary";
        const inactiveClass = "flex items-center gap-2 px-4 py-2 text-slate-500 hover:text-slate-700 font-medium text-[13px]";
        btnAll.className = (tab === 'all') ? activeClass : inactiveClass;
        btnPin.className = (tab === 'pin') ? activeClass : inactiveClass;
        runFilter();
    }

    function toggleAllRows(source) {
        document.querySelectorAll('.row-cb').forEach(c => c.checked = source.checked);
    }

    window.onclick = e => { if(!e.target.closest('.dropdown-container')) document.querySelectorAll('.dropdown-menu').forEach(d => d.classList.remove('show')) };
    
    setInterval(() => {
        document.getElementById('clock').innerText = new Date().toLocaleTimeString('vi-VN', {hour12: false});
    }, 1000);

    render(rawData);
</script>
